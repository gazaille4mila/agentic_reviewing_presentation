% Title slide with full purple background
\newcommand{\milatitleslide}[3]{%
    \begin{frame}[plain]
        \begin{tikzpicture}[remember picture,overlay]
            \fill[milaPurple] (current page.north west) rectangle (current page.south east);
            \node[anchor=north west, xshift=2mm, yshift=5mm] at (current page.north west) {%
                \includegraphics[height=24mm]{\milaWhiteLogoPath}
            };
            \node[anchor=north east, xshift=-10mm, yshift=-9mm] at (current page.north east) {%
                \textcolor{white}{\small\textbf{#2}}
            };
            \draw[white, line width=0.4pt] ([xshift=10mm, yshift=-15mm]current page.north west) --
                                         ([xshift=-10mm, yshift=-15mm]current page.north east);
        \end{tikzpicture}
        \vspace*{30mm}
        \begin{flushleft}
            \textcolor{white}{\Huge #1}
            \vspace{1.2em}

            \textcolor{white}{\large #3}
        \end{flushleft}
    \end{frame}
}

% Section divider slide (full color background, purple title)
\newcommand{\milasectionslide}[4]{%
    \section{#3}%
    \begin{frame}[plain]
        % Pass base color name, not the light version
        \colorlet{milasectionbg}{#1Light}
        \begin{tikzpicture}[remember picture,overlay]
            \fill[milasectionbg] (current page.north west) rectangle (current page.south east);
        \end{tikzpicture}
        \vspace{2cm}
        \begin{flushleft}
            \textcolor{milaPurple}{\small\textbf{#2}}\\[0.5em]
            \textcolor{milaPurple}{\Huge\textbf{#3}}\\[1em]
            \textcolor{milaPurple}{\normalsize #4}
        \end{flushleft}
    \end{frame}
}

% End slide
\newcommand{\milaendslide}[2]{%
    \begin{frame}[plain]
        \begin{tikzpicture}[remember picture,overlay]
            \fill[milaPurple] (current page.north west) rectangle (current page.south east);
        \end{tikzpicture}
        \vspace{2.5cm}
        \begin{flushleft}
            \textcolor{white}{\Huge\textbf{Questions?}}\\[2em]
            \textcolor{milaPurpleLight}{\normalsize #1}\\[1em]
            \textcolor{milaPurpleLight}{\small #2}
        \end{flushleft}
    \end{frame}
}

    colframe=#1,

% User feedback (liked)
\newcommand{\userfeedbacklike}[1]{%
    \begin{feedbackbox}{milaGreen}
        \textcolor{milaGreen}{\faThumbsUp[regular]}\ \ \textbf{Feedback:} #1
    \end{feedbackbox}
}

% User feedback (disliked)
\newcommand{\userfeedbackdislike}[1]{%
    \begin{feedbackbox}{milaRed}
        \textcolor{milaRed}{\faThumbsDown[regular]}\ \ \textbf{Feedback:} #1
    \end{feedbackbox}
